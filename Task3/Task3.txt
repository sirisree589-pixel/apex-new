  <section class="grid-container">
    <!-- Image Carousel -->
    <div>
      <h2>Image Carousel</h2>
      <div class="carousel">
        <img id="carouselImage" src="https://picsum.photos/id/1015/400/300" alt="Carousel Image">
        <div class="carousel-controls">
          <button onclick="prevImage()">⟨ Prev</button>
          <button onclick="nextImage()">Next ⟩</button>
        </div>
      </div>
    </div>

    <!-- API Fetch -->
    <div>
      <h2>Random Joke (API)</h2>
      <div class="joke-box" id="jokeBox">Click the button to load a joke!</div>
      <button onclick="getJoke()">Get Joke</button>
    </div>
  </section>

  <script>
    // === Carousel Logic ===
    const images = [
      'https://picsum.photos/id/1015/400/300',
      'https://picsum.photos/id/1016/400/300',
      'https://picsum.photos/id/1018/400/300',
      'https://picsum.photos/id/1019/400/300'
    ];
    let currentImage = 0;

    function showImage() {
      document.getElementById('carouselImage').src = images[currentImage];
    }

    function nextImage() {
      currentImage = (currentImage + 1) % images.length;
      showImage();
    }

    function prevImage() {
      currentImage = (currentImage - 1 + images.length) % images.length;
      showImage();
    }

    // === API Fetch Logic ===
    async function getJoke() {
      const res = await fetch('https://icanhazdadjoke.com/', {
        headers: { 'Accept': 'application/json' }
      });
      const data = await res.json();
      document.getElementById('jokeBox').textContent = data.joke;
    }
  </script>
